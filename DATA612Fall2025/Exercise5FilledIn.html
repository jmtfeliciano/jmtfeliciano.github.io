# Place your script(s) below:
data("starwars")

tatooine_alderaan_excluded <- starwars |> 
  filter(!homeworld %in% c("Tatooine", "Alderaan")) |>
  select(name, gender, homeworld, species)

write_xlsx(tatooine_alderaan_excluded, "data/TatooineAlderaanExcluded.xlsx")
